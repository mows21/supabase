```sql
-- Vector similarity search function
CREATE OR REPLACE FUNCTION match_embeddings(
    query_embedding VECTOR(1536),
    match_count INT DEFAULT 10,
    filter_type TEXT DEFAULT NULL
)
RETURNS TABLE (
    id UUID,
    content TEXT,
    content_type TEXT,
    similarity FLOAT
)
LANGUAGE plpgsql
AS $$
BEGIN
    RETURN QUERY
    SELECT
        e.id,
        e.content,
        e.content_type,
        1 - (e.embedding <=> query_embedding) AS similarity
    FROM public.embeddings e
    WHERE filter_type IS NULL OR e.content_type = filter_type
    ORDER BY e.embedding <=> query_embedding
    LIMIT match_count;
END;
```

```typescript
import { Langfuse } from 'langfuse';

export class LangfuseObservability {

    return this.langfuse.trace({
      name,
      metadata,
      tags: ['qms-ml-loop', 'supabase-agent'],
    });
  }

  startGeneration(trace: any, name: string, input: unknown) {
    return trace.generation({
      name,
      input,
      model: 'claude-3-opus',
    });
  }

  logSpan(trace: any, name: string, input: unknown, output: unknown) {
    return trace.span({
      name,
      input,
      output,
    });
  }

  logScore(trace: any, name: string, value: number, comment?: string) {
    return trace.score({
      name,
      value,
      comment,
    });
  }

  async flush() {
    await this.langfuse.shutdownAsync();
  }
}
```

```typescript
// utils/parallel-executor.ts
interface AgentTask {
  agentType: string;
  input: unknown;
  priority?: number;
}

interface AgentResult {
  agentType: string;
  output: unknown;
  success: boolean;
  error?: string;
  duration: number;
}

export async function executeParallelAgents(
  tasks: AgentTask[],
  maxConcurrency: number = 5
): Promise<AgentResult[]> {
  const results: AgentResult[] = [];
  const executing: Promise<void>[] = [];

  for (const task of tasks) {
    const promise = executeAgent(task).then(result => {
      results.push(result);
    });

    executing.push(promise);

    if (executing.length >= maxConcurrency) {
      await Promise.race(executing);
      executing.splice(
        executing.findIndex(p => p === promise),
        1
      );
    }
  }

  await Promise.all(executing);
  return results;
}

async function executeAgent(task: AgentTask): Promise<AgentResult> {
  const startTime = Date.now();

  try {
    // Invoke the specific agent edge function
    const response = await fetch(
      `${Deno.env.get('SUPABASE_URL')}/functions/v1/agents/${task.agentType}`,
      {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
          'Authorization': `Bearer ${Deno.env.get('SUPABASE_SERVICE_ROLE_KEY')}`,
        },
        body: JSON.stringify(task.input),
      }
    );

    const output = await response.json();

    return {
      agentType: task.agentType,
      output,
      success: response.ok,
      duration: Date.now() - startTime,
    };
  } catch (error) {
    return {
      agentType: task.agentType,
      output: null,
      success: false,
      error: error.message,
      duration: Date.now() - startTime,
    };
  }
}

// Result merging with deduplication
export function mergeAgentResults(results: AgentResult[]): unknown {
  const successfulResults = results.filter(r => r.success);

  // Structure-first merge
  const merged = {
    insights: [],
    actions: [],
    recommendations: [],
    metadata: {
      totalAgents: results.length,
      successfulAgents: successfulResults.length,
      totalDuration: results.reduce((sum, r) => sum + r.duration, 0),
    },
  };

  for (const result of successfulResults) {
    const output = result.output as any;

    if (output.insights) merged.insights.push(...output.insights);
    if (output.actions) merged.actions.push(...output.actions);
    if (output.recommendations) merged.recommendations.push(...output.recommendations);
  }

  // Deduplication using similarity
  merged.insights = deduplicateByContent(merged.insights);
  merged.recommendations = deduplicateByContent(merged.recommendations);

  return merged;
}

function deduplicateByContent(items: any[]): any[] {
  const seen = new Set<string>();
  return items.filter(item => {
    const key = JSON.stringify(item);
    if (seen.has(key)) return false;
    seen.add(key);
    return true;
  });
}
```

```typescript
// ml-loop/feedback-loop.ts
export class MLFeedbackLoop {
  private supabase: SupabaseClient;
  private langfuse: LangfuseObservability;

  constructor(supabase: SupabaseClient, langfuse: LangfuseObservability) {
    this.supabase = supabase;
    this.langfuse = langfuse;
  }

  // Step 1: Collect feedback from various sources
  async collectFeedback(agentRunId: string, feedback: {
    score: number;
    comment?: string;
    source: 'user' | 'automated' | 'llm-judge';
  }) {
    const { data: run } = await this.supabase
      .from('agents.agent_runs')
      .select('*')
      .eq('id', agentRunId)
      .single();

    if (!run) throw new Error('Agent run not found');

    // Store feedback for training
    await this.supabase.from('ml_loop.training_data').insert({
      model_name: run.agent_type,
      input: run.input,
      output: run.output,
      feedback_score: feedback.score,
      is_validated: feedback.source === 'user',
      source: feedback.source,
    });

    // Log to Langfuse
    this.langfuse.logScore(
      { id: run.langfuse_trace_id },
      'feedback',
      feedback.score,
      feedback.comment
    );
  }

  // Step 2: Analyze patterns in feedback
  async analyzePatterns(modelName: string, timeWindow: string = '7 days') {
    const { data: feedbackData } = await this.supabase
      .from('ml_loop.training_data')
      .select('*')
      .eq('model_name', modelName)
      .gte('created_at', `now() - interval '${timeWindow}'`);

    const analysis = {
      totalSamples: feedbackData?.length || 0,
      averageScore: 0,
      scoreDistribution: {} as Record<number, number>,
      lowScorePatterns: [] as any[],
    };

    if (feedbackData && feedbackData.length > 0) {
      analysis.averageScore =
        feedbackData.reduce((sum, d) => sum + (d.feedback_score || 0), 0) / feedbackData.length;

      // Find patterns in low-scoring outputs
      const lowScoreData = feedbackData.filter(d => (d.feedback_score || 0) < 0.5);
      // Use Claude to analyze patterns (simplified)
      analysis.lowScorePatterns = await this.identifyPatterns(lowScoreData);
    }

    return analysis;
  }

  // Step 3: Detect model drift
  async detectDrift(modelName: string) {
    const { data: recentMetrics } = await this.supabase
      .from('ml_loop.drift_metrics')
      .select('*')
      .eq('model_name', modelName)
      .order('created_at', { ascending: false })
      .limit(100);

    const driftAnalysis = {
      hasDrift: false,
      driftScore: 0,
      affectedMetrics: [] as string[],
      recommendation: '',
    };

    if (recentMetrics && recentMetrics.length > 0) {
      const avgDrift = recentMetrics.reduce((sum, m) => sum + (m.drift_score || 0), 0) / recentMetrics.length;
      driftAnalysis.driftScore = avgDrift;
      driftAnalysis.hasDrift = avgDrift > 0.3; // Threshold

      if (driftAnalysis.hasDrift) {
        driftAnalysis.affectedMetrics = recentMetrics
          .filter(m => (m.drift_score || 0) > 0.3)
          .map(m => m.metric_type);
        driftAnalysis.recommendation = 'Consider retraining with recent data';
      }
    }

    return driftAnalysis;
  }

  // Step 4: Curate training dataset
  async curateDataset(modelName: string, minScore: number = 0.7) {
    const { data: validatedData } = await this.supabase
      .from('ml_loop.training_data')
      .select('*')
      .eq('model_name', modelName)
      .eq('is_validated', true)
      .gte('feedback_score', minScore);

    return {
      datasetSize: validatedData?.length || 0,
      samples: validatedData || [],
      qualityThreshold: minScore,
    };
  }

  // Step 5: Trigger retraining
  async triggerRetrain(modelName: string, config: {
    minSamples: number;
    qualityThreshold: number;
  }) {
    const dataset = await this.curateDataset(modelName, config.qualityThreshold);

    if (dataset.datasetSize < config.minSamples) {
      return {
        triggered: false,
        reason: `Insufficient samples: ${dataset.datasetSize}/${config.minSamples}`,
      };
    }

    // Create new model version
    const newVersion = `v${Date.now()}`;
    await this.supabase.from('ml_loop.model_versions').insert({
      model_name: modelName,
      version: newVersion,
      config: {
        training_samples: dataset.datasetSize,
        quality_threshold: config.qualityThreshold,
      },
      status: 'training',
    });

    // Emit retraining event (could trigger external ML pipeline)
    return {
      triggered: true,
      version: newVersion,
      datasetSize: dataset.datasetSize,
    };
  }

  private async identifyPatterns(data: any[]): Promise<any[]> {
    // Simplified - would use Claude to analyze patterns
    return data.slice(0, 5).map(d => ({
      input_pattern: typeof d.input === 'string' ? d.input.substring(0, 100) : 'complex',
      score: d.feedback_score,
    }));
  }
}
```
